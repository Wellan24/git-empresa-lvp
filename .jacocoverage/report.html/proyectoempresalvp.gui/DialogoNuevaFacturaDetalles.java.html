    private void tPrecioKeyTyped(java.awt.event.KeyEvent evt) {//GEN-FIRST:event_tPrecioKeyTyped

        char car = evt.getKeyChar();
        if(((car < '0' || car > '9') && car != '.') || tPrecio.getText().contains(".") && car == '.') {
            evt.consume();
        }
    }//GEN-LAST:event_tPrecioKeyTyped
